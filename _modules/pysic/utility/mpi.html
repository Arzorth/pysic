[docs]def write_file(lines, filename):
    """Writes the given lines to a text file so that only the master cpu writes.
        
    Parameters:
        
    lines: list of strings
        lines to be written
        
    filename: string
        the name of the file
    """
    if is_master():
        f = file(filename,'w')
        string = ""
        for line in lines:
            string += line+"\n"
        f.write(string[:-2])
        f.close()
